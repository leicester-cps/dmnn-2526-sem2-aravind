import tensorflow as tf
from tensorflow.keras import layers, models
import matplotlib.pyplot as plt
import numpy as np

# ---------------------------------------------------------
# TASK 1: Dataset Analysis
# ---------------------------------------------------------

# Loading CIFAR-10
(train_images, train_labels), (test_images, test_labels) = tf.keras.datasets.cifar10.load_data()

# Normalizing pixel values
train_images, test_images = train_images / 255.0, test_images / 255.0

class_names = ['airplane', 'automobile', 'bird', 'cat', 'deer',
               'dog', 'frog', 'horse', 'ship', 'truck']

# Print dataset stats
print(f"Input Shape: {train_images.shape[1:]}")
print(f"Training samples: {len(train_images)}")
print(f"Test samples: {len(test_images)}")

# Visualizing one example per class
plt.figure(figsize=(10, 5))
for i in range(10):
    idx = np.where(train_labels == i)[0][0]
    plt.subplot(2, 5, i+1)
    plt.xticks([])
    plt.yticks([])
    plt.imshow(train_images[idx])
    plt.xlabel(class_names[i])
plt.tight_layout()
plt.show()

# ---------------------------------------------------------
# TASK 2: Baseline CNN (Minimal Architecture)
# ---------------------------------------------------------

baseline_model = models.Sequential([
    # Single Conv layer for basic feature extraction
    layers.Conv2D(32, (3, 3), activation='relu', input_shape=(32, 32, 3)),
    # Max pooling to reduce dimensionality
    layers.MaxPooling2D((2, 2)),
    # Flattening for the output layer
    layers.Flatten(),
    layers.Dense(10)
])

baseline_model.compile(optimizer='adam',
                      loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),
                      metrics=['accuracy'])

print("\n--- Baseline Model Summary ---")
baseline_model.summary()

# ---------------------------------------------------------
# TASK 3: Extended CNN (Architectural Reasoning)
# ---------------------------------------------------------

extended_model = models.Sequential([
    # Layer 1
    layers.Conv2D(32, (3, 3), activation='relu', input_shape=(32, 32, 3)),
    layers.MaxPooling2D((2, 2)),

    # Extension: Added a second Conv layer and increased filters to 64
    layers.Conv2D(64, (3, 3), activation='relu'),
    layers.MaxPooling2D((2, 2)),

    # Extension: Added Dropout to prevent overfitting
    layers.Dropout(0.2),

    layers.Flatten(),
    layers.Dense(64, activation='relu'),
    layers.Dense(10)
])

extended_model.compile(optimizer='adam',
                      loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),
                      metrics=['accuracy'])

print("\n--- Extended Model Summary ---")
extended_model.summary()

# ---------------------------------------------------------
# TASK 4: Training & Evaluation
# ---------------------------------------------------------

# Training both models
print("\nTraining Baseline...")
history_baseline = baseline_model.fit(train_images, train_labels, epochs=10,
                                    validation_split=0.2, verbose=1)

print("\nTraining Extended...")
history_extended = extended_model.fit(train_images, train_labels, epochs=10,
                                    validation_split=0.2, verbose=1)

# Plotting results
def plot_metrics(history, name):
    acc = history.history['accuracy']
    val_acc = history.history['val_accuracy']
    loss = history.history['loss']
    val_loss = history.history['val_loss']
    epochs = range(1, len(acc) + 1)

    plt.figure(figsize=(12, 4))

    plt.subplot(1, 2, 1)
    plt.plot(epochs, loss, 'bo-', label='Train Loss')
    plt.plot(epochs, val_loss, 'ro-', label='Val Loss')
    plt.title(f'{name} Loss')
    plt.legend()

    plt.subplot(1, 2, 2)
    plt.plot(epochs, acc, 'bo-', label='Train Acc')
    plt.plot(epochs, val_acc, 'ro-', label='Val Acc')
    plt.title(f'{name} Accuracy')
    plt.legend()
    plt.show()

plot_metrics(history_baseline, "Baseline")
plot_metrics(history_extended, "Extended")

# ---------------------------------------------------------
# TASK 5: Test Set Evaluation
# ---------------------------------------------------------

print("\nFinal Performance on Test Set:")
b_loss, b_acc = baseline_model.evaluate(test_images, test_labels, verbose=0)
e_loss, e_acc = extended_model.evaluate(test_images, test_labels, verbose=0)

print(f"Baseline Test Accuracy: {round(b_acc, 4)}")
print(f"Extended Test Accuracy: {round(e_acc, 4)}")
