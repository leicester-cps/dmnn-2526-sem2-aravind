import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import PolynomialFeatures, StandardScaler
from sklearn.pipeline import Pipeline
from sklearn.metrics import mean_squared_error, mean_absolute_error, r2_score

# Set seed for reproducibility
np.random.seed(42)

# 1. Load the Data
# Ensure the file is in the same directory as your script
data_path = 'california_housing_train.csv'
df = pd.read_csv(data_path)

# 2. Prepare Features and Target
# Using 'median_income' for simple regression as per lab instructions
X_simple = df[['median_income']]
# Using all features except the target for multiple regression
X_multi = df.drop('median_house_value', axis=1)
y = df['median_house_value']

# Split the data (80% train, 20% test)
X_train_s, X_test_s, y_train, y_test = train_test_split(X_simple, y, test_size=0.2, random_state=42)
X_train_m, X_test_m, _, _ = train_test_split(X_multi, y, test_size=0.2, random_state=42)

# --- SECTION 1: SIMPLE LINEAR REGRESSION ---
print("--- Simple Linear Regression (Median Income vs Value) ---")
simple_model = LinearRegression()
simple_model.fit(X_train_s, y_train)
y_pred_s = simple_model.predict(X_test_s)

print(f"R2 Score: {r2_score(y_test, y_pred_s):.4f}")
print(f"MAE: ${mean_absolute_error(y_test, y_pred_s):,.2f}\n")
